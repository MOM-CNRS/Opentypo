Mon Application document.addEventListener('DOMContentLoaded', function() { // Détecter l'item actif basé sur l'URL const currentPath = window.location.pathname; const menuItems = document.querySelectorAll('.sidebar-menu-item'); menuItems.forEach(function(item) { const link = item.querySelector('a') || item; const href = link.getAttribute('href') || link.getAttribute('action') || ''; // Vérifier si l'URL correspond à l'item if (href && (currentPath.includes(href.split('?')[0]) || (currentPath === '/' && href.includes('index.xhtml')) || (currentPath.endsWith('/') && href.includes('index.xhtml')))) { item.classList.add('active'); } // Gérer les clics pour ajouter la classe active link.addEventListener('click', function() { // Retirer active de tous les items menuItems.forEach(function(menuItem) { menuItem.classList.remove('active'); }); // Ajouter active à l'item cliqué item.classList.add('active'); }); }); // Setup tooltips pour les items du menu setupSidebarMenuTooltips(); }); function setupSidebarMenuTooltips() { const menuItems = document.querySelectorAll('.sidebar-menu-item'); menuItems.forEach(function(item) { const tooltip = item.querySelector('.sidebar-menu-item-tooltip'); const textElement = item.querySelector('.sidebar-menu-item-text'); if (!tooltip && textElement) { const tooltipText = textElement.textContent.trim(); if (tooltipText) { const tooltipSpan = document.createElement('span'); tooltipSpan.className = 'sidebar-menu-item-tooltip'; tooltipSpan.textContent = tooltipText; item.querySelector('.sidebar-menu-item-content').appendChild(tooltipSpan); } } }); }
